		else

			newnode=newnode->next;
			newnode->prev=temp;
			newnode->data=val;

void display(dnode *head)
{
       dnode *temp=head;
       while(temp!=NULL)
       {
	printf("\t%d",temp->data);
	temp=temp->next;
	}
}
