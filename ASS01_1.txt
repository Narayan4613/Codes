/* stack: reverse string using array*/
#include<stdio.h>
#include<conio.h>
#include<string.h>
void main()
{
char stk[10],str[10];
int top=-1,i=0,j=0;
int len;

clrscr();
printf("\n Enter String:");
scanf("%s",str);
len=strlen(str);

for(i=0;i<=len-1;i++)
{ char ch=str[i];
  top++;
  stk[top]=ch;

}
for(i=0;i<=len-1;i++)
{  str[j]=stk[top];
   top--;
   j++;
}
printf("\n Reverse String is:%s",str);
getch();
}
