/* insertion sort-ascending*/

#include<stdio.h>
#include<conio.h>
void main()
{ int a[]={12,5,8,40,2},i,j,pivot,pos;
  clrscr();

  printf("\n Array: ");
  for(i=0;i<5;i++)
  printf("%3d",a[i]);

  for(i=1;i<5;i++)
  {  pivot=a[i];
     pos=i;
     for(j=i-1;j>=0;j--)
     {    if(a[j]>pivot)
	{    a[j+1]=a[j];
		     pos--;
	}
     }
   a[pos]=pivot;
  }
  printf("\n\n Sorted array: ");
	for(i=0;i<5;i++)
	printf("%3d",a[i]);
getch();   }
