/* function:polynomial evaluation */
#include<stdio.h>
#include<conio.h>
#include<math.h>
#define MAX 100

void main()
{ int a[MAX],i,n;
  char sign;
float x,sum=0;
float eval(int [],float,int); // array,base x ,size n
clrscr();
printf("\n Enter order of polynomial:");
scanf("%d",&n);

printf("\n Enter the coefficient into array:");
for(i=n;i>=0;i--)
scanf("%d",&a[i]);

printf("\n Enter the value of x:");
scanf("%f",&x);

printf("\n Polynomial Representation :");
for(i=n;i>=0;i--)
{  if(a[i]>=0) // positive coeff
   sign='+';
   else       // negative coeff
   sign='-';

if(i>1) // check power
	printf("%c %dx^%d ",sign,abs(a[i]),i);
else if(i==1)
	printf("%c %dx ",sign,abs(a[i]));
else
	printf("%c %d ",sign,abs(a[i]));
}
sum = eval(a,x,n); // fn call

printf("\n\n Value of polynomial: %f",sum);
getch();
}
float eval(int a[], float x, int n) // fn definition
{ float ans=0;
  int i;

for(i=n;i>=0;i--)
ans=ans+pow(x,i)*a[i];

return ans;
}
