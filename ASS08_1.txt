/* DMA: 1-D (array) table of integers */
#include<stdio.h>
#include<conio.h>

void main()
{ int *a,i,n,cnt,j=0;
clrscr();

printf("\n Enter n to print table form 1 to n:");
scanf("%d",&n);
a=(int *)malloc(n*10*(sizeof(int)));
for(cnt=1;cnt<=n;cnt++)
{ for(i=1;i<=10;i++)
 { a[j]=cnt*i;
   j++;
 }
}
for(j=0;j<n*10;j++)
{
  if(j%10==0)
  printf("\n------------\n");

  printf("\n%d",a[j]);
}
getch();
}
